country_event = {
	id = devolved_random_events.1

	hide_window = yes
    is_triggered_only = yes
    trigger = {
        is_country_type = default
        has_civic = devolved_civic_randivist_principles
    }

    immediate = {
		
		random_list = {
			33 = {set_variable = {which = "effectduration" value = 5}}
			33 = {set_variable = {which = "effectduration" value = 3}}
		    33 = {set_variable = {which = "effectduration" value = 1}}
			1 = {set_variable = {which = "effectduration" value = 25}}
		}

		random_list = {
			80 = {set_variable = {which = "multipliercount" value = 2}}
			9 = {set_variable = {which = "multipliercount" value = 1}}
			9 = {set_variable = {which = "multipliercount" value = 4}}
			2 = {set_variable = {which = "multipliercount" value = 20}}
		}

        add_modifier = {
			modifier = random_event_modifier_A_1
			multiplier = multipliercount
			years = 3
		}
	}     
}
